# DevNote v0.3.0 — Recent Notes Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add local SQLite storage and a Recent Notes sidebar list so DevNote remembers every synced note and users can browse/re-read past notes without leaving VS Code.

**Architecture:** One new file (`src/MemoryStore.ts`) wraps sql.js and owns all database logic. SidebarProvider gets three new responsibilities: save-on-sync, refresh-on-idle, handle-click-on-recent. The webview gains a Recent Notes list in the Idle state and a "Clear all memory" popup in Settings. NotionService.push() is updated to return page metadata.

**Tech Stack:** sql.js (WASM SQLite), crypto.randomUUID() for IDs, vanilla JS webview, TypeScript, VS Code extension API.

### Task 2: Create MemoryStore with schema migration

**Files:**
- Create: `src/MemoryStore.ts`

This is the core new file. It wraps sql.js, manages the `devnote.db` lifecycle, runs migrations, and exposes a typed API. No other file in the codebase imports sql.js directly.

- [ ] **Step 1: Create `src/MemoryStore.ts` with types, init, and migration logic**

```typescript
import * as vscode from 'vscode';
import * as fs from 'fs';
import * as path from 'path';
import initSqlJs, { Database as SqlDatabase } from 'sql.js';
import { StructuredNote } from './LLMService';

export interface RecentNoteRow {
  id: string;
  title: string;
  branchName: string;
  createdAt: number;
}

export interface FullNote {
  id: string;
  title: string;
  branchName: string;
  content: StructuredNote;
  contentMarkdown: string;
  notionPageId: string | null;
  notionPageUrl: string | null;
  createdAt: number;
}

export interface NoteMetadata {
  branchName: string;
  notionPageId: string;
  notionPageUrl: string;
}

const MIGRATIONS: { [version: number]: (db: SqlDatabase) => void } = {
  1: (db) => {
    db.run(`
      CREATE TABLE notes (
        id               TEXT PRIMARY KEY,
        title            TEXT NOT NULL,
        branch_name      TEXT NOT NULL,
        content_json     TEXT NOT NULL,
        content_markdown TEXT NOT NULL,
        notion_page_id   TEXT,
        notion_page_url  TEXT,
        created_at       INTEGER NOT NULL,
        embedding        BLOB,
        embedding_model  TEXT
      )
    `);
    db.run(`CREATE INDEX idx_notes_created_at ON notes(created_at DESC)`);
    db.run(`
      CREATE TABLE schema_version (
        version    INTEGER PRIMARY KEY,
        applied_at INTEGER NOT NULL
      )
    `);
    db.run(`INSERT INTO schema_version (version, applied_at) VALUES (1, ?)`, [Date.now()]);
  },
};

export class MemoryStore {
  private db: SqlDatabase | null = null;
  private dbPath: string;
  private wasmPath: string;
  private _available = false;

  get available(): boolean {
    return this._available;
  }

  constructor(context: vscode.ExtensionContext) {
    const globalDir = context.globalStorageUri.fsPath;
    this.dbPath = path.join(globalDir, 'devnote.db');
    // sql.js WASM file is in node_modules inside the extension install directory
    this.wasmPath = path.join(context.extensionUri.fsPath, 'node_modules', 'sql.js', 'dist');
  }

  async init(): Promise<void> {
    try {
      // Ensure the globalStorage directory exists
      const dir = path.dirname(this.dbPath);
      if (!fs.existsSync(dir)) {
        fs.mkdirSync(dir, { recursive: true });
      }

      const SQL = await initSqlJs({
        locateFile: (file: string) => path.join(this.wasmPath, file),
      });

      if (fs.existsSync(this.dbPath)) {
        try {
          const fileBuffer = fs.readFileSync(this.dbPath);
          this.db = new SQL.Database(fileBuffer);
        } catch {
          // DB file is corrupted — rename and start fresh
          const timestamp = Date.now();
          const corruptPath = `${this.dbPath}.corrupt-${timestamp}`;
          fs.renameSync(this.dbPath, corruptPath);
          vscode.window.showWarningMessage(
            'Your DevNote memory was corrupted and has been reset. The old file was saved as a backup.'
          );
          this.db = new SQL.Database();
        }
      } else {
        this.db = new SQL.Database();
      }

      this.runMigrations();
      await this.persist();
      this._available = true;
    } catch (err) {
      this._available = false;
      const detail = err instanceof Error ? err.message : String(err);
      console.error(`[DevNote] MemoryStore init failed: ${detail}`);
    }
  }

  private runMigrations(): void {
    if (!this.db) return;

    let currentVersion = 0;
    try {
      const result = this.db.exec('SELECT MAX(version) as version FROM schema_version');
      if (result.length > 0 && result[0].values.length > 0 && result[0].values[0][0] !== null) {
        currentVersion = result[0].values[0][0] as number;
      }
    } catch {
      // schema_version table doesn't exist yet — version is 0
      currentVersion = 0;
    }

    const versions = Object.keys(MIGRATIONS).map(Number).sort((a, b) => a - b);
    for (const v of versions) {
      if (v > currentVersion) {
        MIGRATIONS[v](this.db);
      }
    }
  }

  async saveNote(note: StructuredNote, meta: NoteMetadata): Promise<string> {
    if (!this.db || !this._available) {
      throw new Error('MemoryStore is not available');
    }

    const id = crypto.randomUUID();
    const contentJson = JSON.stringify(note);
    const contentMarkdown = this.serializeNoteToMarkdown(note);
    const createdAt = Date.now();

    this.db.run(
      `INSERT INTO notes (id, title, branch_name, content_json, content_markdown,
                          notion_page_id, notion_page_url, created_at)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
      [id, note.title, meta.branchName, contentJson, contentMarkdown,
       meta.notionPageId, meta.notionPageUrl, createdAt]
    );

    await this.persist();
    return id;
  }

  async getRecentNotes(): Promise<RecentNoteRow[]> {
    if (!this.db || !this._available) return [];

    const result = this.db.exec(
      'SELECT id, title, branch_name, created_at FROM notes ORDER BY created_at DESC'
    );

    if (result.length === 0) return [];

    return result[0].values.map((row) => ({
      id: row[0] as string,
      title: row[1] as string,
      branchName: row[2] as string,
      createdAt: row[3] as number,
    }));
  }

  async getNoteById(id: string): Promise<FullNote | null> {
    if (!this.db || !this._available) return null;

    const result = this.db.exec('SELECT * FROM notes WHERE id = ?', [id]);
    if (result.length === 0 || result[0].values.length === 0) return null;

    const row = result[0].values[0];
    // Column order: id, title, branch_name, content_json, content_markdown,
    //               notion_page_id, notion_page_url, created_at, embedding, embedding_model
    return {
      id: row[0] as string,
      title: row[1] as string,
      branchName: row[2] as string,
      content: JSON.parse(row[3] as string) as StructuredNote,
      contentMarkdown: row[4] as string,
      notionPageId: (row[5] as string) || null,
      notionPageUrl: (row[6] as string) || null,
      createdAt: row[7] as number,
    };
  }

  async clearAll(): Promise<void> {
    if (!this.db || !this._available) {
      throw new Error('MemoryStore is not available');
    }
    this.db.run('DELETE FROM notes');
    await this.persist();
  }

  async exportAll(): Promise<string> {
    if (!this.db || !this._available) return '[]';

    const result = this.db.exec(
      `SELECT id, title, branch_name, content_json, content_markdown,
              notion_page_id, notion_page_url, created_at
       FROM notes ORDER BY created_at DESC`
    );

    if (result.length === 0) return '[]';

    const notes = result[0].values.map((row) => ({
      id: row[0],
      title: row[1],
      branchName: row[2],
      content: JSON.parse(row[3] as string),
      contentMarkdown: row[4],
      notionPageId: row[5],
      notionPageUrl: row[6],
      createdAt: row[7],
    }));

    return JSON.stringify(notes, null, 2);
  }

  private serializeNoteToMarkdown(note: StructuredNote): string {
    return [
      `# ${note.title}`,
      '',
      '## Summary',
      note.summary,
      '',
      '## What Changed',
      ...note.whatChanged.map((c) => `- ${c}`),
      '',
      '## Why',
      note.why,
      '',
      '## Key Decisions',
      note.keyDecisions,
      '',
      '## Files Affected',
      ...note.filesAffected.map((f) => `- ${f}`),
    ].join('\n');
  }

  private async persist(): Promise<void> {
    if (!this.db) return;
    try {
      const data = this.db.export();
      const buffer = Buffer.from(data);
      fs.writeFileSync(this.dbPath, buffer);
    } catch (err) {
      const detail = err instanceof Error ? err.message : String(err);
      vscode.window.showWarningMessage(
        `Couldn't save memory to disk. Notes are safe in this session but may not survive a restart. (${detail})`
      );
    }
  }
}
```
